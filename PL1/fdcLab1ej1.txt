#Frecuencia absoluta y frecuencia absoluta acumulada

#para calcular la frecuencia relativa y la frecuencia relativa acumulada
debemos crear funciones.
frecrel<-function(radio){table(radio)/length(radio)}
frecrelacum<-function(Radio){cumsum(table(radio))/length(radio)}

#las guardamos en nuestro workspace para poder utilizarlas de nuevo en caso de necesidad
dump("frecrel",file="frecrel.R")
dump("frecrelacum",file="frecrelacum.R")
